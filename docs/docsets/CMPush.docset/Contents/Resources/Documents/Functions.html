  • CMPush_SetApplePushEnvironment(_:)

    Configure CMPush. Should be called at application startup to set the apple push environment. Production or Sandbox. Default value is production

    Example

    import CMPush

...

func application(_ application: UIApplication, didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {
   // Configure CMPush
#if DEBUG
   CMPush_SetApplePushEnvironment(.Sandbox)
#endif
   CMPush_Config(applicationKey: "<appkey>", groupName: "group.com.cm.myapp")

   UIApplication.shared.registerForRemoteNotifications()
   UNUserNotificationCenter.current().delegate = self
   return true
}

            UIApplication.shared.open(url, options: [:], completionHandler: nil)

  • CMPush_UpdateMSISDN(_:withResult:)

    Update MSISDN. This triggers an OTP sms, so use with care. Only when new MSISDN is entered or resend is needed.

    Example

    import CMPush

...
CMPush_UpdateMSISDN(msisdn) { result in
    switch result {
    case .success:
        //go to otp screen, for example:
        self.performSegue(withIdentifier: "optcode", sender: nil)
    case .failure(let error):
        var errorMessage : String?
        switch error {
        case .notRegistered:
            //Make sure CM_UpdateToken was completed successfully
            break
        case .noMsisdn:
            //Cannot happen here
            break
        case .offline(let error):
            errorMessage = (error as NSError).localizedFailureReason ?? "No internet connection"
        case .serverError(message: let message, details: let details):
            errorMessage = message
        case .systemError(error: let error):
            print(error)
        case .internalError(message: let message):
            print(message)
        @unknown default:
            break
        }
        if let message = errorMessage {
            let alert = UIAlertController(title: "Error", message: message, preferredStyle: .alert)
            alert.addAction(UIAlertAction(title: "OK", style: .default, handler: nil))
            self.present(alert, animated: true)
        }
    }
}

  • CMPush_HandleNotificationRequest(_:withContentHandler:)

    Pre-process the received CM Push NotificationRequest. This function reports the received status to CM to let CM verify the delivery of the push message. This function also handles media-attachments in the notificationrequest. It will pre-download media and attach it to the Push Notification.

    If the CM Push message contains suggestions they are added here as buttons to the Push Notification. Supported Action types: CMPushSuggestion

    Add the following code to your UNNotificationServiceExtension.

    import CMPush

...

override func didReceive(_ request: UNNotificationRequest, withContentHandler contentHandler: @escaping (UNNotificationContent) -> Void) {
   CMPush_HandleNotificationRequest(request, withContentHandler: contentHandler)
}
