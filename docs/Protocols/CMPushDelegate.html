CMPushDelegate

public protocol CMPushDelegate : AnyObject

Delegate protocol that is called by the SDK when the user selected a suggestion (button).

  • cmPushHandleSuggestion(_:)

    Called by the SDK when the user selected a suggestion (button). Handle in the app as fits.

    Example

    extension AppDelegate : CMPushDelegate {
    func cmPushHandleSuggestion(_ suggestion: CMPushSuggestion) {
        switch suggestion {
        case .openURL(let label, let url):
         UIApplication.shared.open(url, options: [:], completionHandler: nil)
        case .openAppPage(label: let label, let page):
         print("Open app page: \(page)")
        @unknown default:
         break
        }
    }
}
